CREATE TABLE location (
 id UUID PRIMARY KEY,
 name VARCHAR(20), 
 city VARCHAR(20) NOT NULL UNIQUE, 
 district VARCHAR(20), 
 village VARCHAR(20), 
 zipcode INT NOT NULL,
 street VARCHAR(25),
 CHECK (city_id IS NOT NULL OR village IS NOT NULL OR DISTRICT IS NOT NULL)
);

CREATE TABLE localities (
 id SERIAL PRIMARY KEY,
 name VARCHAR(25), 
 type 
);
